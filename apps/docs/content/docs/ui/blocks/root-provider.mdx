import { TypeTable } from 'next-docs-ui/components/type-table'

The context provider of all the components, including `next-themes` and context
for search dialog. It should be located at the root layout.

## Usage

```jsx
import { RootProvider } from 'next-docs-ui/provider'

export default function Layout({ children }) {
  return (
    <html lang="en">
      <body>
        <RootProvider>{children}</RootProvider>
      </body>
    </html>
  )
}
```

### Search Dialog

You can customize the search dialog from the root provider.

<TypeTable
  type={{
    SearchDialog: {
      description: 'Replace the default dialog',
      type: 'React.FC<SearchDialogProps>'
    },
    links: {
      type: '[string, string][]',
      description: 'Default items to show (Only supported by default dialog)'
    }
  }}
/>

```jsx
<RootProvider
  search={{
    links: [
      ['Home', '/'],
      ['Docs', '/docs']
    ]
  }}
>
  ...
</RootProvider>
```
